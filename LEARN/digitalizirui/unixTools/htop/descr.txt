# commands
t	tree "command"
m	sort by Memory
p	sort by Processor
/procNm	find process by name
\	reverce filter
\+esc	disable reverce filter
u+user	show particular user processes
k	kill process
e	shows environment variables that are around process
"space"	select multiple lines
U	reset selection
F2	settings


# check what files uses process:
sudo apt install lsof

# then find process in htop
l	shows files used by the process
